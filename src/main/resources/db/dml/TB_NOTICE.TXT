CREATE TABLE `TB_NOTICE` (
	`NOTICE_ID` VARCHAR(18) NOT NULL,
	`CRE_USER` VARCHAR(18) NOT NULL,
	`CRE_DATETIME` DATETIME NOT NULL,
	`UPD_SER` VARCHAR(18) NULL DEFAULT NULL,
	`UPD_DATETIME` DATETIME NULL DEFAULT NULL,
	`NOTICE_TITLE` VARCHAR(200) NOT NULL,
	`NOTICE_CONTENT` VARCHAR(4000) NOT NULL,
	`RETRIEVE_CNT` INT(255) NULL DEFAULT '0',
	`DEL_YN` CHAR(1) NULL DEFAULT 'N'
)
COMMENT='게시판'
ENGINE=InnoDB
